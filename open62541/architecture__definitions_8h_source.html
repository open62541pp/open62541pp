358UA_atomic_xchg(void * volatile * addr, void *newptr) {
359#if UA_MULTITHREADING >= 100
360#ifdef _MSC_VER /* Visual Studio */
361 return _InterlockedExchangePointer(addr, newptr);
362#else /* GCC/Clang */
363 return __sync_lock_test_and_set(addr, newptr);
364#endif
365#else
366 void *old = *addr;
367 *addr = newptr;

369#endif
