32/**
33 * Subscription parameters with default values from open62541.
34 *
35 * Parameters are passed by reference because illegal parameters can be revised by the server.
36 * The updated parameters reflect the actual values that the server will use.
37 */
38struct SubscriptionParameters {
39 /// Cyclic interval in milliseconds that the subscription is requested to return notifications.
40 double publishingInterval = 500.0;
41 /// Delete the subscription after defined publishing cycles without sending any notifications.
42 uint32_t lifetimeCount = 10000;
43 /// Send keep-alive message after defined publishing cycles without notifications.
44 uint32_t maxKeepAliveCount = 10;
45 /// The maximum number of notifications per publish (0 if unlimited).
46 uint32_t maxNotificationsPerPublish = 0;
47 /// Relative priority of the subscription. Notifications with higher priority are sent first.
48 uint8_t priority = 0;
49};
