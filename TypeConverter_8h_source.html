205template <typename T, TypeIndex... typeIndexes>
206struct TypeConverterNative {
207 using ValueType = T;
208 using NativeType = T;
209 using ValidTypes = TypeIndexList<typeIndexes...>;
210
211 static_assert(ValidTypes::size() >= 1);
212
213 static void fromNative(const T& src, T& dst) {
214 if constexpr (std::is_fundamental_v<T>) {
215 dst = src;
216 } else {
217 // just take first type -> underlying memory layout of all types should be the same
218 constexpr auto typeIndexGuess = ValidTypes::toArray().at(0);
219 // clear first
220 UA_clear(&dst, getUaDataType<typeIndexGuess>());
221 // deep copy
222 const auto status = UA_copy(&src, &dst, getUaDataType<typeIndexGuess>());
223 throwOnBadStatus(status);
224 }
225 }
226
227 static void toNative(const T& src, T& dst) {
228 fromNative(src, dst);
229 }
230};
